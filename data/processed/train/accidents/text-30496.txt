հունիսի ին ժամը ին արտակարգ իրավիճակների նախարարություն ահազանգ է ստացվել որ ստեփանավան տաշիր ավտոճանապարհին տեղի է ունեցել ճտպ կա տուժած դեպքի վայր է մեկնել մեկ մարտական հաշվարկ պարզվել է որ լոռու մարզի ստեփանավան քաղաքի վիրահայոց մայրուղու ին կմ ում բախվել են ֆորդ մակնիշի սս պետհամարանիշով միկրոավտուբուսը վարորդ լ բ ծնվ թ եւ վազ մակնիշի լօ պետհամարանիշով ավտոմեքենան վարորդ հ վ ծնվ թ ինչի հետեւանքով հ վ ը տարբեր աստիճանի մարմնական վնասվածքներով հոսպիտալացվել է ստեփանավանի բժշկական կենտրոն որտեղ բժիշկները տուժածի առողջական վիճակը գնահատել են բավարար հրշեջ փրկարարները հոսանքազրկել են վազ մակնիշի ավտոմեքենան եւ փակել են գազի բալոնի փականը
