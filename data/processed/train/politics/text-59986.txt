այն որ այս որոշումը բոլորի համար սպասելի էր պարոն մուկուչյանը վստահաբար չի սխալվել մենք արդեն տարվա պատմություն ունենք սահմանադրական դատարանի կողմից ընտրական վեճերով արդարադատության իրականացման այդ մասին ազատություն ռադիոկայանի հետ զրույցում նշել է փաստաբան սդ ում կոնգրես հժկ դաշինքի ներկայացուցիչ վահե գրիգորյանը հիշեցնենք որ դաշինքը սդ ում վիճարկում էր աժ ընտրությունների արդյունքները եւ սդ ն մերժեց դաշինքի հայցը փաստորեն մի կուսակցություն որը վերահսկում է գործադիր օրենսդիր եւ դատական իշխանությունները միաժամանակ վերահսկում է նաեւ սահմանադրական դատարանի դահլիճում ձեր եւ հանրության կողմից տեսած հակ եւ հժկ կուսակցություններից զատ մնացած բոլոր կողմերին կենտրոնական ընտրական հանձնաժողով ոստիկանություն դատախազություն եվ այդ շրջանակներում խոսել այն մասին որ ապացույցներ չեն ներկայացվել այստեղ արդեն պարոն մուկուչյանի ասվածի որեւէ իմաստ կամ բովանդակություն չի մնում սահմանադրական դատարանը պիտի փոխի իր դիրքորոշումն ընտրական վեճերի վերաբերյալ ապացուցման պարտականության բեռի նժարը պետք է էապես փոխվի դեպի պատասխանող կողմը պատասխանող կողմն է որ վերահսկում է ամբողջ ռեսուրսը տիրապետում է ողջ օպերատիվ հետախուզական ինֆորմացիային հսկայական պետական ռեսուրս եվ եթե ընտրությունների օրինականությունը ընտրությունների օրինակարգությունը պիտի ճշտվի այն տեստով թե արդյոք հակ հժկ ն կարողացավ ապացուցել ընտրությունների անօրինականությունը թե ոչ եւ եթե չկարողացավ ապացուցել անօրինականությունը ուրեմն ընտրություններն օրինական են սա ըստ էության գրեթե անհնար պարտականություն է ռադիոկայանի հետ զրույցում ասել է գրիգորյանը
