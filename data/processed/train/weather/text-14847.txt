հհ տարածքում հունիսի ը շիրակում լոռիում եւ տավուշում օրվա երկրորդ կեսին սպասվում է կարճատեւ անձրեւ եւ ամպրոպ քամին հարավ արեւելյան մ վրկ ամպրոպի ժամանակ հնարավոր է քամու ուժգնացում մ վրկ օդի ջերմաստիճանը հունիսի ը աստիճանաբար կբարձրանա աստիճանով երեւանում հունիսի ը սպասվում է առանց տեղումների եղանակ երեւանի հունիսի ի բազմամյա կլիմայական տվյալներ օդի առավելագույն ջերմաստիճան թ օդի նվազագույն ջերմաստիճան թ օդի միջին օրական ջերմաստիճան
