աստծո առաջ մենք բոլորս հավասար ենք ունենք հավասար իրավունքներ եւ առավելեւս այսօր աշխարհի շատ երկրներում մենք տեսնում ենք թե կանայք ինչ հսկայական դերակատարություն ունեն հասարակական տնտեսական կյանքում կանայք հանուն խաղաղության միջոցառման ժամանակ իր ելույթում ասաց կոնգրես հժկ դաշինքի պատգամավորության թեկնածու զոյա թադեւոսյանը նա նշեց որ հայ ընտանիքում հայ կինը ամենասիրված էակն է եւ երեխա ժամանակ եւ հետագայում որպես ընտանիքի մայր կարծում եմ որ մենք այսօր մեծ պատասխանատվություն ունենք մեր պետության կայացման գործում որովհետեւ մեր պետությունն այսօր կառավարվում է ապաշնորհ անկարգապահ անկազմակերպ տղամարդու կողմից հետեւաբար կանայք պետք է գան ուղղելու այդ սխալը եւ կյանքի կոչելու այնպիսի ծրագրեր որոնք հայ մարդուն կթողնեն իր հայրենիքում իսկ դա կարող է անել միայն մայրը իսկ դա կարող է անել միայն սեր ունեցող արարածը դա կինն է ամենամեծ սիրով տոգորված կանայք եկե ք խաղաղություն բերենք մեր զավակներին միակ ժառանգությունը որ մենք կարող ենք նրանց տալ խաղաղությունն է ասաց թադեւոսյանը լիլիթ թադեւոսյան մանրամասները տեսանյութում
